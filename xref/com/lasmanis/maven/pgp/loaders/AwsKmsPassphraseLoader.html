16  package com.lasmanis.maven.pgp.loaders;
17  
18  import com.lasmanis.maven.pgp.loaders.helpers.AwsCryptoHelper;
19  import com.lasmanis.maven.pgp.loaders.helpers.CryptoHelper;
20  import org.apache.maven.plugin.MojoExecutionException;
21  import org.bouncycastle.openpgp.PGPSecretKey;
22  import org.codehaus.plexus.component.annotations.Component;
23  import org.codehaus.plexus.component.annotations.Requirement;
24  import org.kohsuke.maven.pgp.PassphraseLoader;
25  import org.kohsuke.maven.pgp.PgpMojo;
26  import java.io.IOException;
27  import java.util.Map;
28  
29  /**
30   * Decryption wrapper for a pass phrase via AWS KMS.
31   *
32   * @author mpl
33   */
34  @Component(role = PassphraseLoader.class, hint = "awskms")
35  public final class AwsKmsPassphraseLoader
36      extends PassphraseLoader {
37      /**
38       * all the loaders that Plexus knows about.
39       * */
40      @Requirement(role = PassphraseLoader.class)
41      private Map<String, PassphraseLoader> loaders;
42  
43      /**
44       * Our crypto helper.
45       */
46      private final CryptoHelper crypto;
47  
48      /**
49       * Constructor.
50       */
51      public AwsKmsPassphraseLoader() {
52          super();
53          this.crypto = new AwsCryptoHelper();
54      }
55  
56      /**
57       * Constructor for unit testing
58       *
59       * This constructors allows the map of loaders to
60       * be initialized by the test harness.  This is normally
61       * handled directly by Plexus.
62       *
63       * @param loaders the Map of loaders
64       * @param crypto the crypto helper
65       */
66      AwsKmsPassphraseLoader(
67              final Map<String, PassphraseLoader> loaders,
68              final CryptoHelper crypto) {
69          super();
70          this.loaders = loaders;
71          this.crypto = crypto;
72      }
73  
74      @Override
75      /** {@inheritDoc} */
76      public String load(
77              final PgpMojo pm,
78              final PGPSecretKey pgpsk,
79              final String string)
80              throws IOException,
81                  MojoExecutionException {
82          // check
83          if (string == null || string.isEmpty()) {
84              throw new MojoExecutionException(
85                      "Scheme 'awskms:' expects an additional specifier.");
86          }
87  
88          // make sure we have crypto support
89          if (this.crypto == null) {
90              throw new MojoExecutionException(
91                      "Failed to initialize the crypto library.");
92          }
93  
94          // extract the scheme
95          int head = string.indexOf(':');
96          if (head < 0) {
97              throw new MojoExecutionException(
98                      "Invalid additional specifier. It needs to start with a "
99                      + "scheme like 'FOO:': " + string);
100         }
101         String scheme = string.substring(0, head);
102 
103         // get the loader
104         PassphraseLoader loader = null;
105         if (this.loaders != null) {
106             loader = this.loaders.get(scheme);
107         }
108         if (loader == null) {
109             throw new MojoExecutionException(
110                     "Invalid passphrase scheme '" + scheme
111                     + "'. If this is your custom scheme, perhaps you forgot "
112                     + "to specify it in <dependency> to this plugin?");
113         }
114 
115         // get the cipherText
116         String cipherText = loader.load(pm, pgpsk, string.substring(head + 1));
117 
118         // decrypt the pass phrase
119         return this.crypto.decrypt(cipherText);
120     }
121 }
